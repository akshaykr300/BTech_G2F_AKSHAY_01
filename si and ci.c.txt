#include <stdio.h>
#include <math.h>
int main() {
float p,roi,t,si,amount,ci;

printf("enter principle amount\n");
scanf("%f",&p);
printf("enter rate of interest\n");
scanf("%f",&roi);
printf("enter time period\n");
scanf("%f",&t);

si=(p*roi*t)/100;
amount=p*pow((1+roi/100),t);
ci=amount-p;

printf("the si is %f",si);
printf("the compound interest is %f",ci);


return 0;

}
